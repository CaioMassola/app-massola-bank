import React, { useState } from 'react';
import { View, Text, Image, Alert } from 'react-native';
import { Background } from '../../components/Background';
import Logo from '../../assets/svg/Vector.svg'
import { Button, Input } from 'react-native-elements';

import { styles } from './styles';
import { font } from '../../global/font';
import { ScrollView, TouchableOpacity } from 'react-native-gesture-handler';
import { ButtonIcon } from '../../components/ButtonIcon';
import { ListDivider } from '../../components/ListDivider';
import { SafeAreaView } from 'react-native-safe-area-context';

const google = require('../../assets/google.jpg');

export const Login = () => {

  const [email, setEmail] = useState<string>('');
  const [password, setPassword] = useState<string>('');
  const [errorEmail, setErroEmail] = useState<boolean>(false);
  const [errorPassword, setErroPassword] = useState<boolean>(false);
  const [visibilityPassword, setVisibilityPassword] = useState<boolean>(false);

  const changeVisibilityPassword = () => {
    if (!visibilityPassword) {
      setVisibilityPassword(true);
    } else {
      setVisibilityPassword(false);
    }
  }

  const handleSignValidate = (mail: string, pass: string) => {

    if (mail == '' && pass != '') {
      setErroEmail(true);
    }

    if (pass == '' && email != '') {
      setErroPassword(true);
    }

    if (pass == '' && mail == '') {
      setErroEmail(true);
      setErroPassword(true)
    }

    if (mail != '') {
      let valid = mail.indexOf("@")
      let valid2 = mail.indexOf(".com")

      if (valid < 0 || valid2 < 0) {
        setErroEmail(true)
        return
      }
    }

    if (email && password) {
      handleSign()
    }
  }

  const handleSign = () => {

    let dados = {
      email: email,
      password: password
    }

    console.log(dados)

    if (dados.email != 'caio@caio.com' || dados.password != 'caio123') {
      Alert.alert('Ops!', 'Email ou senha inválidos!');
      setErroEmail(false);
      setErroPassword(false);
    } else {
      Alert.alert('Sucesso!', 'Aguarde um instante para entrar!')
    }
  }

  return (
    <Background>
      <SafeAreaView>
        <ScrollView>
          <View style={styles.container}>
            <View style={styles.viewIcon}>
              <Logo />
            </View>
            <Text style={styles.texto}>Bem-Vindo</Text>
            <View style={styles.inputView}>
              <Input
                placeholder='email...'
                inputStyle={{ color: 'white', fontFamily: font.fonts.title400 }}
                placeholderTextColor={`${errorEmail ? '#ff4d4d' : 'white'}`}
                keyboardType={'email-address'}
                rightIcon={{ type: 'material', name: 'email', color: 'white' }}
                errorStyle={{ color: '#ff4d4d', fontSize: 16, fontFamily: font.fonts.title400 }}
                errorMessage={errorEmail ? 'Digite um email válido!' : ''}
                onChangeText={setEmail}
                onKeyPress={() => setErroEmail(false)}
              />
              <Input
                placeholder='senha...'
                inputStyle={{ color: 'white', fontFamily: font.fonts.title400 }}
                placeholderTextColor={`${errorPassword ? '#ff4d4d' : 'white'}`}
                secureTextEntry={!visibilityPassword ? true : false}
                rightIcon={{
                  type: 'material', name: `${!visibilityPassword ? 'visibility-off' : 'visibility'}`, color: 'white', onPress: () => {
                    changeVisibilityPassword()
                  }
                }}
                errorStyle={{ color: '#ff4d4d', fontSize: 16, fontFamily: font.fonts.title400 }}
                errorMessage={errorPassword ? 'Digite uma senha válida!' : ''}
                onChangeText={setPassword}
                onKeyPress={() => setErroPassword(false)}
              />
            </View>
            <View style={{ alignItems: 'center' }}>
              <View style={styles.buttonView}>
                <ButtonIcon
                  color='white'
                  type='material'
                  icon='login'
                  height={40}
                  onPress={() => handleSignValidate(email, password)}
                  title={'Entrar'}
                  img={0} />
              </View>
            </View>
            <TouchableOpacity
              onPress={() => console.log('criarconta')}
            >
              <Text style={styles.createAnAccount}>Não possui uma conta ainda?</Text>
            </TouchableOpacity>
            <View style={{ marginTop: 25 }}>
              <ListDivider />
            </View>
            <View style={styles.googleView}>
              <Text style={styles.textGoogle}>Ou entre com </Text>
              <View style={{ alignItems: 'center' }}>
                <View style={styles.buttonView}>
                  <ButtonIcon
                    color='white'
                    img={google}
                    height={40}
                    onPress={() => console.log('Google')}
                    title={'Google'}
                    icon={''}
                  />
                </View>
              </View>
            </View>
          </View>
        </ScrollView>
      </SafeAreaView>
    </Background>
  )
}
