import React, { useState } from 'react';
import { Input } from 'react-native-elements';
import { View, Text } from 'react-native';
import { SafeAreaView } from 'react-native-safe-area-context';
import { ScrollView, TouchableOpacity } from 'react-native-gesture-handler';

import { Background } from '../../components/Background';
import { ButtonIcon } from '../../components/ButtonIcon';
import { ListDivider } from '../../components/ListDivider';

import Logo from '../../assets/svg/Vector.svg'
import { font } from '../../global/font';
import { styles } from './styles';

import { useForm, Controller } from 'react-hook-form';
import * as yup from 'yup';

const google = require('../../assets/google.jpg');

export const Login = () => {

  const fieldsValidationSchema = yup.object().shape({
    email: yup
    .string()
    .required('O email é obrigatório!')
    .email('Digite um email válido'),
    password: yup
    .string()
    .required('A senha é obrigatória!')
    .min(6, 'A senha deve conter pelo menos 6 dígitos')
  })

  const [visibilityPassword, setVisibilityPassword] = useState<boolean>(false);
  const { control, handleSubmit, formState: { errors}, register, setValue } = useForm({context: fieldsValidationSchema})

  const onSubmit = (data: any) => {
    console.log(data)
  }

  const changeVisibilityPassword = () => {
    !visibilityPassword ? setVisibilityPassword(true) : setVisibilityPassword(false);
  }


  return (
    <Background>
      <SafeAreaView>
        <ScrollView>
          <View style={styles.container}>
            <View style={styles.viewIcon}>
              <Logo />
            </View>
            <Text style={styles.texto}>Bem-Vindo</Text>
            <View style={styles.inputView}>
              <Controller
                name="email"
                control={control}
                render={(props) => (
                  <Input
                    placeholder='email...'
                    inputStyle={{ color: 'white', fontFamily: font.fonts.title400 }}
                    keyboardType={'email-address'}
                    rightIcon={{ type: 'material', name: 'email', color: 'white' }}
                    errorStyle={{ color: '#ff4d4d', fontSize: 16, fontFamily: font.fonts.title400 }}
                    errorProps={errors?.email}
                    onChangeText={(value) => {
                      props.field.onChange(value)
                    }}
                  />
                )}
              />
              <Controller
                name="password"
                control={control}
                render={(props) => (
                  <Input
                    placeholder='senha...'
                    inputStyle={{ color: 'white', fontFamily: font.fonts.title400 }}
                    secureTextEntry={!visibilityPassword ? true : false}
                    rightIcon={{
                      type: 'material', name: `${!visibilityPassword ? 'visibility-off' : 'visibility'}`, color: 'white', onPress: () => {
                        changeVisibilityPassword()
                      }
                    }}
                    errorStyle={{ color: '#ff4d4d', fontSize: 16, fontFamily: font.fonts.title400 }}
                    errorProps={errors?.password}
                    onChangeText={(value) => {
                      props.field.onChange(value)
                    }}
                  />
                )}
              />
            </View>
            <View style={{ alignItems: 'center' }}>
              <View style={styles.buttonView}>
                <ButtonIcon
                  color='white'
                  type='material'
                  icon='login'
                  height={40}
                  onPress={handleSubmit(onSubmit)}
                  title={'Entrar'}
                  img={0} />
              </View>
            </View>
            <TouchableOpacity
              onPress={() => console.log('criarconta')}
            >
              <Text style={styles.createAnAccount}>Não possui uma conta ainda?</Text>
            </TouchableOpacity>
            <View style={{ marginTop: 25 }}>
              <ListDivider />
            </View>
            <View style={styles.googleView}>
              <Text style={styles.textGoogle}>Ou entre com </Text>
              <View style={{ alignItems: 'center' }}>
                <View style={styles.buttonView}>
                  <ButtonIcon
                    color='white'
                    img={google}
                    height={40}
                    onPress={() => console.log('Google')}
                    title={'Google'}
                    icon={''}
                  />
                </View>
              </View>
            </View>
          </View>
        </ScrollView>
      </SafeAreaView>
    </Background>
  )
}
